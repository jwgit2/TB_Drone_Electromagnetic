CREATE TABLE IF NOT EXISTS MEASUREMENTS(
ID_MEASUREMENT_SET INTEGER,
MEASUREMENT_DATETIME DATETIME, MEASUREMENT_MS INTEGER, COMMENT TEXT,
RANGE_VM INTEGER, RESISTOR REAL,
TEMPERATURE REAL, PRESSURE REAL, HUMIDITY REAL,
VOLTAGE_EFM REAL, ELECTRIC_FIELD REAL, VOLTAGE_INA REAL);

CREATE TABLE IF NOT EXISTS POSITION(
ID_MEASUREMENT_SET INTEGER,
MEASUREMENT_DATETIME DATETIME, MEASUREMENT_MS INTEGER, COMMENT TEXT,
POSITION_X REAL, POSITION_Y REAL, POSITION_Z REAL);
